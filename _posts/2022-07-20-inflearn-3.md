## Solution

```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.ArrayList;
import java.util.List;
import java.util.StringTokenizer;

class Main {
    static List<String> returnWinner(int N, int[] informationA, int[] informationB) {
        List<String> list = new ArrayList<>();

        // 1: 가위, 2: 바위, 3: 보
        for (int i = 0; i < N; i++) {
            if (informationA[i] == 1) {
                if (informationB[i] == 1) {
                    list.add("D");
                } else if (informationB[i] == 2) {
                    list.add("B");
                } else {
                    list.add("A");
                }
            } else if (informationA[i] == 2) {
                if (informationB[i] == 1) {
                    list.add("A");
                } else if (informationB[i] == 2) {
                    list.add("D");
                } else {
                    list.add("B");
                }
            } else { // informationA[i] == 3
                if (informationB[i] == 1) {
                    list.add("B");
                } else if (informationB[i] == 2) {
                    list.add("A");
                } else {
                    list.add("D");
                }
            }
        }

        return list;
    }

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        int N = Integer.parseInt(br.readLine());

        StringTokenizer st1 = new StringTokenizer(br.readLine(), " ");
        int[] informationA = new int[N];
        for (int i = 0; i < informationA.length; i++) {
            informationA[i] = Integer.parseInt(st1.nextToken());
        }

        StringTokenizer st2 = new StringTokenizer(br.readLine(), " ");
        int[] informationB = new int[N];
        for (int i = 0; i < informationB.length; i++) {
            informationB[i] = Integer.parseInt(st2.nextToken());
        }

        for (String x : returnWinner(N, informationA, informationB)) {
            System.out.println(x);
        }

        br.close();
    }
}
```

```java
import java.util.Scanner;

class Main {
    public String solution(int n, int[] a, int[] b) {
        String answer = "";
        for (int i = 0; i < n; i++) {
            if (a[i] == b[i]) {
                answer += "D";
            } else if (a[i] == 1 && b[i] == 3) {
                answer += "A";
            } else if (a[i] == 2 && b[i] == 1) {
                answer += "A";
            } else if (a[i] == 3 && b[i] == 2) {
                answer += "A";
            } else {
                answer += "B";
            }
        }
        return answer;
    }

    public static void main(String[] args) {
        Main T = new Main();
        Scanner kb = new Scanner(System.in);
        int n = kb.nextInt();
        int[] a = new int[n];
        int[] b = new int[n];
        for (int i = 0; i < n; i++) {
            a[i] = kb.nextInt();
        }
        for (int i = 0; i < n; i++) {
            b[i] = kb.nextInt();
        }
        for (char x : T.solution(n, a, b).toCharArray()) {
            System.out.println(x);
        }
    }
}
```
