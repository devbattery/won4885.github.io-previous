## Solution

```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.StringTokenizer;

class Main {
    static int returnSumOfMax(int N, int[][] arrN) {
        int max = Integer.MIN_VALUE;

        // 1. 각 행의 합
        for (int i = 0; i < N; i++) {
            int tmpRow = 0;
            for (int j = 0; j < N; j++) {
                tmpRow += arrN[i][j];

                if (tmpRow > max) {
                    max = tmpRow;
                }
            }
        }

        // 2. 각 열의 합
        for (int i = 0; i < N; i++) {
            int tmpColumn = 0;
            for (int j = 0; j < N; j++) {
                tmpColumn += arrN[j][i];
            }

            if (tmpColumn > max) {
                max = tmpColumn;
            }
        }

        // 3. 두 대각선의 합
        int tmpDiagonal = 0;
        int tmpDiagonalReverse = 0;
        for (int i = 0; i < N; i++) {
            tmpDiagonal += arrN[i][i];
            tmpDiagonalReverse += arrN[i][N - i - 1];

            if (tmpDiagonal > max) {
                max = tmpDiagonal;
            }
            if (tmpDiagonalReverse > max) {
                max = tmpDiagonal;
            }
        }

        return max;
    }

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        int N = Integer.parseInt(br.readLine());
        int[][] arrN = new int[N][N];
        for (int i = 0; i < N; i++) {
            StringTokenizer st = new StringTokenizer(br.readLine(), " ");
            for (int j = 0; j < N; j++) {
                arrN[i][j] = Integer.parseInt(st.nextToken());
            }
        }
        System.out.print(returnSumOfMax(N, arrN));
        br.close();
    }
}
```

```java
import java.util.Scanner;

class Main {
    public int solution(int n, int[][] arr) {
        int answer = Integer.MIN_VALUE;
        int sum1, sum2;
        for (int i = 0; i < n; i++) {
            sum1 = sum2 = 0;
            for (int j = 0; j < n; j++) {
                sum1 += arr[i][j];
                sum2 += arr[j][i];
            }
            answer = Math.max(answer, sum1);
            answer = Math.max(answer, sum2);
        }

        sum1 = sum2 = 0;
        for (int i = 0; i < n; i++) {
            sum1 += arr[i][i];
            sum2 += arr[i][n - i - 1];
        }
        answer = Math.max(answer, sum1);
        answer = Math.max(answer, sum2);

        return answer;
    }

    public static void main(String[] args) {
        Main T = new Main();
        Scanner kb = new Scanner(System.in);
        int n = kb.nextInt();
        int[][] arr = new int[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                arr[i][j] = kb.nextInt();
            }
        }

        System.out.println(T.solution(n, arr));
    }
}
```
